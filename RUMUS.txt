INSERT INTO SKOR_PEGAWAI(ID_PEGAWAI,POINT)
SELECT POINT_PEGAWAI.`ID_PEGAWAI`,POINT_PEGAWAI.`point` FROM POINT_PEGAWAI

UPDATE SKOR_PEGAWAI SET ID_INCOME = 'INC/V/2018'

UPDATE SKOR_PEGAWAI SET JML_HADIR = (SELECT DAY(LAST_DAY( NOW() ) ) - ABSEN );

UPDATE SKOR_PEGAWAI SET POINT_AWAL = (POINT*JML_HADIR) / (SELECT DAY(LAST_DAY( NOW() ) ));


UPDATE MSTR_INCOME
JOIN SKOR_PEGAWAI ON MSTR_INCOME.`ID_INCOME` = SKOR_PEGAWAI.`ID_INCOME`
SET MSTR_INCOME.SUM_POINT_AWAL = (SELECT SUM(SKOR_PEGAWAI.`POINT_AWAL`) FROM SKOR_PEGAWAI)
WHERE SKOR_PEGAWAI.`ID_INCOME` = MSTR_INCOME.`ID_INCOME`

UPDATE SKOR_PEGAWAI
JOIN MSTR_INCOME ON MSTR_INCOME.`ID_INCOME` = SKOR_PEGAWAI.`ID_INCOME`
SET SKOR_PEGAWAI.POINT_AKHIR = (MSTR_INCOME.`SUM_POINT_AWAL`/SKOR_PEGAWAI.POINT_AWAL)
WHERE SKOR_PEGAWAI.`ID_INCOME` = MSTR_INCOME.`ID_INCOME`


UPDATE SKOR_PEGAWAI
JOIN MSTR_INCOME ON MSTR_INCOME.`ID_INCOME` = SKOR_PEGAWAI.`ID_INCOME`
SET SKOR_PEGAWAI.NOMINAL = (MSTR_INCOME.`JUMLAH`) / (SKOR_PEGAWAI.POINT_AKHIR)
WHERE SKOR_PEGAWAI.`ID_INCOME` = MSTR_INCOME.`ID_INCOME`
